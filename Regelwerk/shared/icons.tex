% Zentrale Symbolbefehle fuer Heldenpfade
% SVG-Quelldateien liegen unter Regelwerk/assets/icons.
% Der Build konvertiert sie vor dem LaTeX-Lauf nach assets/icons/generated.

% Einheitliche Einbindung mit sauberer Ausrichtung an der Textgrundlinie.
\newcommand{\hpIcon}[2][1em]{%
  \raisebox{-0.14em}{\includegraphics[height=#1]{assets/icons/generated/#2.pdf}}%
}

\newcommand{\hpActionIcon}[1]{%
  \raisebox{0.1ex}{\textcircled{\sffamily\bfseries\scriptsize #1}}%
}
