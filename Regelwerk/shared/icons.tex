% Zentrale Symbolbefehle fuer Heldenpfade

% Einheitliche Einbindung mit sauberer Ausrichtung an der Textgrundlinie.

  \raisebox{-0.14em}{\includesvg[height=#1]{assets/icons/#2}}%
}

  \raisebox{0.1ex}{\textcircled{\sffamily\bfseries\scriptsize #1}}%
}
